
```yaml
# 1. 보안 및 기본 서버 설정
auth_enabled: false  # 멀티테넌시(사용자별 권한 분리) 비활성화. 내부망 단일 사용 환경에 적합.
server:
  http_listen_port: 3100
  grpc_listen_port: 9096
  log_level: debug     # 상세 로그를 남겨 초기 구축 시 트러블슈팅에 유리함.

# 2. 공통 경로 설정 (데이터가 저장될 뿌리 경로)
common:
  instance_addr: 127.0.0.1
  path_prefix: /data/log/mcash/kfcs_json_api/loki
  storage:
    filesystem:
      chunks_directory: /data/log/mcash/kfcs_json_api/loki/chunks
      rules_directory: /data/log/mcash/kfcs_json_api/loki/rules
  replication_factor: 1
  ring:
    kvstore:
      store: inmemory  # 단일 노드 구성이므로 클러스터용 저장소 대신 메모리 사용.

# 3. 데이터 저장 구조 (Schema)
schema_config:
  configs:
    - from: 2020-10-24
      store: tsdb       # 최신 인덱스 저장 방식인 TSDB 사용 (성능과 용량 효율이 좋음)
      object_store: filesystem
      schema: v13       # 최신 스키마 버전 적용
      index:
        prefix: index_
        period: 24h     # 인덱스 파일을 24시간 단위로 생성하여 관리.

# 4. 데이터 수집 및 메모리 관리 (Ingester)
ingester:
  wal:
    enabled: true       # Write Ahead Log 활성화. 서버 급정거 시 메모리 데이터 복구 가능.
    dir: /data/log/mcash/kfcs_json_api/loki/wal
  chunk_idle_period: 30s    # 30초 동안 새 로그가 없는 청크는 즉시 디스크로 보냄 (실시간성 향상).
  chunk_retain_period: 1m   # 디스크로 보낸 후 1분간은 메모리에 유지 (빠른 재조회 대비).
  chunk_target_size: 2048576 # 청크가 약 2MB가 되면 디스크로 보냄.

# 5. 데이터 자동 삭제 (Retention)
compactor:
  working_directory: /data/log/mcash/kfcs_json_api/loki/compactor
  shared_store: filesystem
  retention_enabled: true   # 오래된 로그 자동 삭제 기능 활성화.
  retention_delete_delay: 2h
  retention_delete_worker_count: 2

# 6. 유입량 제한 및 보관 주기 (핵심 운영 설정)
limits_config:
  ingestion_rate_mb: 8      # 초당 최대 유입량 8MB 제한 (서버 부하 방지).
  ingestion_burst_size_mb: 16 # 순간 최대 유입 허용량 16MB.
  retention_period: 120h    # 로그 보관 주기: 5일 (120시간). 저장 용량 확보를 위한 전략적 수치.
  max_query_length: 3h       # 한 번에 조회 가능한 시간 범위 3시간으로 제한 (무거운 쿼리 방지).
```
