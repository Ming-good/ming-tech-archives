@Slf4j  
@Configuration  
@RequiredArgsConstructor  
public class PartitionThreadJobConfig {  
  
    private final JobRepository repository;  
    private final PlatformTransactionManager transactionManager;  
    private final DataSource dataSource;  
    private final ThreadPartitioner partitioner;  
  
    @Bean  
    public Job partitionJob() {  
        return new JobBuilder("partitionJob", repository)  
                .start(managerStep())  
                .incrementer(new RunIdIncrementer())  
                .build();  
    }  
  
    @Bean  
    public Step managerStep() {  
        return new StepBuilder("managerStep", repository)  
                .partitioner("workerStep",partitioner)  
                .step(workerStep())  
                .taskExecutor(threadExecutor2())  
                .gridSize(3)  
                .build();  
    }  
  
    @Bean  
    public Step workerStep() {  
        return new StepBuilder("workerStep", repository)  
                .<ThreadJobConfig.UserDto, ThreadJobConfig.UserDto>chunk(5, transactionManager)  
                .reader(jdbcPartitionThreadReader(null, null))  
                .processor(partitionpProcessor())  
                .writer(consoleWriter())  
                .build();  
    }  
  
    @Bean  
    @StepScope    public ItemWriter<ThreadJobConfig.UserDto> consoleWriter() {  
        return chunk -> {  
            log.info(">>>> Writing items for thread {}", Thread.currentThread().getName());  
            for (ThreadJobConfig.UserDto userDto : chunk.getItems()) {  
                log.info("  >> {}", userDto);  
            }  
        };  
    }  
  
    @Bean  
    public TaskExecutor threadExecutor2() {  
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();  
        executor.setCorePoolSize(3);  
        executor.setMaxPoolSize(3);  
        executor.setWaitForTasksToCompleteOnShutdown(true);  
        executor.setAwaitTerminationSeconds(10);  
        executor.setThreadNamePrefix("MING-9-");  
        executor.setAllowCoreThreadTimeOut(true);  
        executor.setKeepAliveSeconds(30);  
        return executor;  
    }  
  
    @Bean  
    @StepScope    public JdbcPagingItemReader<ThreadJobConfig.UserDto> jdbcPartitionThreadReader(  
            @Value("#{stepExecutionContext['start']}") String start,  
            @Value("#{stepExecutionContext['end']}") String end  
    ) {  
        return new JdbcPagingItemReaderBuilder<ThreadJobConfig.UserDto>()  
                .name("jdbcUserThreadReader2")  
                .dataSource(dataSource)  
                .pageSize(2)  
                .selectClause("select user_id,name, email, address")  
                .fromClause("from users")  
                .whereClause("user_id between " + start + " and " + end + "")  
                .sortKeys(Map.of("user_id", Order.ASCENDING))  
                .beanRowMapper(ThreadJobConfig.UserDto.class)  
                .build();  
    }  
  
    @Bean  
    @StepScope    public ItemProcessor<ThreadJobConfig.UserDto, ThreadJobConfig.UserDto> partitionpProcessor() {  
        return user -> {  
            String threadName = Thread.currentThread().getName();  
            log.info("[{}] {}", threadName, user.getName());  
            return user;  
        };  
    }  
  
    @Data  
    public static class UserDto {  
        private String user_id;  
        private String name;  
        private String email;  
        private String address;  
    }  
}

@Slf4j  
@JobScope  
@Component  
public class ThreadPartitioner implements Partitioner {  
  
    @Override  
    public Map<String, ExecutionContext> partition(int gridSize) {  
        log.info("===== 파티션 생성 =====");  
        ExecutionContext context = new ExecutionContext();  
        context.put("start", "0");  
        context.put("end", "50");  
  
        ExecutionContext context2 = new ExecutionContext();  
        context2.put("start", "100");  
        context2.put("end", "140");  
  
        ExecutionContext context3 = new ExecutionContext();  
        context3.put("start", "150");  
        context3.put("end", "170");  
        return Map.of("1", context, "2", context2, "3", context3);  
    }  
}
