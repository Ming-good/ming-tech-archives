```
public static void main(String[] args) {  
	MyCall call1 = (int value) -> {return value * 2};  
	
	// 단일 표현식인 경우 중괄호와 리턴 생략 가능
	MyCall call2 = (int value) -> value * 2; // 기본  

	// 타입 추론
	MyCall call3 = (value) -> value * 2; // 타입 생략  

	// 매개변수의 괄호 생략
	MyCall call4 = value -> value * 2; // 매개변수가 1개일 때 () 생략 가능  
	  
	System.out.println(call4.call(50));  
}  
interface MyCall {  
	int call(int value);  
}
```
