```kotlin
fun main() {  
    readFun("이민규", 6)  
    readFun("이민규")  
    defaultFun("민규")  
    defaultFun("이민규", target="성공해서 주변 사람들을 행복하게 해주는 것")  
    varFun("이민규", "가", "나", "다")  
}  
// 오버로딩  
fun readFun(name:String, age:Int) {  
    println("안녕하세요 $name 입니다. 나이는 ${age}입니다.")  
}  
fun readFun(name:String) {  
    println("안녕하세요 $name 입니다")  
}  

// 디폴트 매개변수  
fun defaultFun(name:String, age:Int = 33, target:String="부자 되기") {  
    println("안녕하세요 $name 입니다. 나이는 $age 이고 저의 목표는 $target 입니다.")  
}  
  
// vararg (다른 파라미터랑 쓴다면 마지막에 써야 된다.)  
fun varFun(name:String, vararg args:String) {  
    for (a in args) {  
        println("= 이름: $name || $a")  
    }  
}
```
