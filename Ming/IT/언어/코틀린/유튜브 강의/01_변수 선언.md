```kotlin
// var(일반적인 변수 읽기,쓰기 가능),  
// val(선언시에만 초기화 가능 중간에 값 변경 불가)  
var a: Int = 4;  
val b: Int = 5;  
println(a)  
println(b)  
  
// 코틀린은 null을 허용하지 않으며 값을 입력하지 않는 것을 허용 하지 않는다  
// null을 입력하고 싶다면? Nullable변수  
var aa:Int? = null;  
  
// 10진수, 16진수, 2진수 **8진수는 지원하지 않는다  
var intValue:Int = 1234  
var longValue:Long = 1234L  
var intValueByHex:Int = 0x1af  
var intValueByBin:Int = 0b1010101  
  
// 소수  
var doubleValue:Double = 123.5  
var doubleValueWithExp: Double = 123.5e10  
var flatValue:Float = 123.5f  
  
// 코틀린은 내부적으로 UTF-16BE를 사용한다 2byte이다 한글자당  
// 코틀린은 특수문자 \t, \b, \r, \n. \'. \"등을 지원한다  
var char = 'a'  
var str = "hellow\nmoney"  
println(str)  
  
// 논리형  
var booleanValue:Boolean = true;  
  
// 문자열  
var StringValue = "one line Helloo"  
var multiLineStringBalue = """  
    one line    two line    three line""".trimIndent()  
println(multiLineStringBalue)
```
