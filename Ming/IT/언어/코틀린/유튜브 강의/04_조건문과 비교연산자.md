// Any는 어떤 자료형이든 사용가능한 최상위 자료형이다  
fun doWhen(a:Any) {  
    when(a) {  
        1 -> println("정수 1입니다")  
        "DiMo" -> println("코틀린입니다")  
        is Long -> println("Long입니다")  
        !is String -> println("String 타입이 아닙니다.")  
    }  
}  
  
// 값을 그대로 쓸경우 값을 리턴한다.  
fun doWhen2(a:Any): Any {  
    var result = when(a) {  
        1 -> "정수 1입니다"  
        "DiMo" -> "코틀린입니다"  
        is Long -> 18  
        !is String -> "String 타입이 아닙니다."  
        else -> "asdasdasdasd"  
    }  
    return result  
}
