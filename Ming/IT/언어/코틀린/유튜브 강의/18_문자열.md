```kotlin
fun main() {  
    controllMain()  
    isStrMain()  
}  

fun controllMain() {  
    val str = "Kotlin.Test.Ming"  
    println(str.lowercase(Locale.getDefault()))  
    println(str.uppercase(Locale.getDefault()))  
  
    val split = str.split(".");  
    println(split)  
  
    val joinToString = split.joinToString("-")  
    println(joinToString)  
  
    println(str.substring(5..10))  
}  
  
fun isStrMain() {  
    var nullableStr:String? = null  
    var emptyStr = ""  
    var blankStr = " "  
    var str = "A"  
  
    println(nullableStr.isNullOrEmpty())  
    println(emptyStr.isNullOrEmpty())  
    println(blankStr.isNullOrEmpty())  
    println(str.isNullOrEmpty())  
  
    println(nullableStr.isNullOrBlank())  
    println(emptyStr.isNullOrBlank())  
    println(blankStr.isNullOrBlank())  
    println(str.isNullOrBlank())  
}
```
