```kotlin
fun main() {  
    // 함수를 전달하는 방법  
    b(::a)  
  
    // 람다함수 간소화  
    var c: (String) -> Unit = {str -> println("$str 람다함수")}  
    b(c)  
  
    // 람다함수 참조 간소화  
    var d = {str:String -> println("$str 람다함수")}  
    b(d)  
}  
  
fun a (str: String) {  
    println("$str 함수 a")  
}  
  
fun b(function: (String)->Unit) {  
    function("b가 호출함")  
}
```

## 함수 참조
- `::a` a라는 함수를 가르키는 주소값을 넘김
