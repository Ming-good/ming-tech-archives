```kotlin
fun main() {  
    UsingGeneric(A()).doShouting()  
    UsingGeneric(B()).doShouting()  
    UsingGeneric(C()).doShouting()  
  
    doShouting(B())  
}  
  
fun <T:A> doShouting(t:T) {  
    t.shout()  
}  
  
open class A {  
    open fun shout() {  
        println("A가 소리칩니다")  
    }  
}  
  
class B : A() {  
    override fun shout() {  
        println("B가 소리칩니다")  
    }  
}  
  
class C : A() {  
    override fun shout() {  
        println("C가 소리칩니다")  
    }  
}  
// <T:A>와 같은 의미이다.  
class UsingGeneric<out T> (var t: A) {  
    fun doShouting() {  
        t.shout()  
    }  
}
```
