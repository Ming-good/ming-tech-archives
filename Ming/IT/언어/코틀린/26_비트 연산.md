```kotlin
var bitData: Int = 0b10000  
println((1 shl 2).toString(2))  
  
var orBitData = bitData or(1 shl 2)  
println(orBitData.toString(2))  
  
var andBitData = bitData and(1 shl 4)  
println(andBitData.toString(2))  
  
println(andBitData shr 2)  
  
println(bitData.toString(2))  
var invBitData = bitData.inv() and 31  
println(invBitData.toString(2))  
  
println((orBitData xor(0b11010)).toString(2))
```
