```kotlin
fun main() {  
    val a = Drink()  
    a.drink()  
  
    var b: Drink = Cola()  
    b.drink()  
  
    // 다운 캐스팅 (조건문 안에서만 다운캐스팅 됨)  
    if (b is Cola) {  
        b.washDishes()  
    }  
  
    // as를 사용할 경우 반환값 뿐만 아니라 변수 자체도 다운캐스팅 된다.  
    var c = b as Cola  
    c.washDishes()  
    b.washDishes()  
}  
  
open class Drink {  
    var name = "음료"  
    open fun drink() {  
        println("${name}를 마십니다")  
    }  
}  
  
class Cola: Drink() {  
    var type = "콜라"  
    override fun drink() {  
        println("${name}중에 ${type}를 마십니다")  
    }  
  
    fun washDishes() {  
        println("${type}로 설거지를 합니다")  
    }  
}
```
