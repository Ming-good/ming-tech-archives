## 규칙
1. 서브 클래스는 수퍼 클래스에 존재하는 속성과 같은 이름의 속성을 가질 수 없음
2. 서브 클래스가 생성될 때는 반드시 수퍼클래스의 생성자까지 호출된다
```kotlin
fun main() {  
    var a = Animal("해", 5, "개")  
    var b = Dog("주", 5)  
    var c = Cat("바", 5)  
  
    a.introduce()  
    b.introduce()  
    b.bark()  
    c.introduce()  
    c.meow()  
}  
  
open class Animal (var name:String, var age:Int, var type:String) {  
    fun introduce() {  
        println("저는 ${name}입니다. ${age}살이고 ${type}입니다.")  
    }  
}  
  
class Dog(name:String, age:Int) : Animal(name,age,"Dog") {  
    fun bark() {  
        println("멍멍")  
    }  
}  
  
class Cat(name:String, age:Int) : Animal(name,age,"Cat") {  
    fun meow() {  
        println("미용미용")  
    }  
}
```
