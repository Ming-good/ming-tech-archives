fun main(){  
    var a = 0

        print( a++)

    do {  
        print(a++)  
    } while(a < 10)  
  
    println("증가")  
    for (i in 0..9) {  
        print(i)  
    }  
  
    println("증가 step")  
    for (i in 0..9 step 3) {  
        print(i)  
    }  
  
    println("감소")  
    for (i in 9 downTo 0) {  
        print(i)  
    }  
  
    println("감소 stpe")  
    for (i in 9 downTo 0 step 3) {  
        print(i)  
    }  
  
    println("문자열 for문")  
    for (i in 'a' .. 'e') {  
        print(i)  
    }  
}
