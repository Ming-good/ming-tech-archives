## 문법
#### 조건문
```lua
local a = 3

if a == 3 then
    print("안녕하세요" .. " '..'은 문자열 연결 연산자입니다.")
    local items = {"a", "b", "c", "d"}
    print(#items .. " 개의 아이템이 있습니다. (#은 길이 연산자)")
elseif a == 4 then
    print("a는 4입니다.")
else
    print("a는 3도 4도 아닙니다.")
end
```

#### 함수
```lua
-- 익명 함수를 변수에 할당
local testFunc = function(a, b)
    print("익명 함수 호출됨")
    return 123
end

-- 고차 함수 (함수를 인자로 받는 함수)
function executeFunc(num1, num2, fun)
    print("전달받은 함수를 실행합니다.")
    print("결과값:", fun(num1, num2))
end

executeFunc(1, 2, testFunc)
```

```lua
-- 기본 배열 순회
local arr = {10, 20, 30, 40, 50}
for i = 1, #arr do
    print(i .. "번째 값: " .. arr[i] .. " (루아는 1부터 시작!)")
end

-- 복합 테이블 (배열 + 사전 형식)
local myTable = {"aaa", 1, 2, name = 'Gemini', age = 12}

-- pairs: 모든 키-값 쌍 순회 (사전 형식 포함)
print("--- pairs 순회 ---")
for k, v in pairs(myTable) do
    print(k, v)
end

-- ipairs: 숫자 인덱스만 순차적으로 순회
print("--- ipairs 순회 ---")
for i, v in ipairs(myTable) do
    print(i, v)
end
```
